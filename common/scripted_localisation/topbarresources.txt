defined_text = {
	# global
	name = oil_text
	text = {
		trigger = {
			check_variable = {
				resource_imported@oil>0
			}
			check_variable = {
				resource@oil>20
			}
		}
		localization_key = "[?resource@oil|*0Y]"
	}
	text = {
		trigger = {
			always = yes
		}
		localization_key = "[?resource@oil|*0+]"
	}
}
