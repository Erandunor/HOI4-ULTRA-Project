country_event = {    ### AI China needs more cheap troops ###
	id = hip_ai_troops.1
	title = hip_ai_troops.1.t
	desc = hip_ai_troops.1.d
	
	trigger = {
		original_tag = CHI
		has_war = yes
		has_manpower > 300000
		is_ai = yes
		NOT = {
			controls_state = 802
			controls_state = 608
		}
	}
	
	fire_only_once = yes
	
	immediate = {
		add_manpower = -262400
		load_oob = "CHI_AI_troops"
	}
	
	option = {
	
	}
}
